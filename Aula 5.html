    import pandas as pd
        import numpy as np
        import matplotlib.pyplot as plt
        import seaborn as sns
        
        # Gerando um dataset fictício com valores numéricos e alguns outliers
        np.random.seed(42)
        data = {
            "ID": range(1, 101),
            "Idade": np.random.randint(18, 60, 100),
            "Salario": np.append(np.random.normal(5000, 1500, 95), [30000, 35000, 40000, 45000, 50000]),
            "Horas_Trabalho": np.random.randint(20, 60, 100)
        }
        
        df = pd.DataFrame(data)
        
        # Função para identificar outliers usando o método do IQR
        def detectar_outliers_iqr(df, coluna):
            Q1 = df[coluna].quantile(0.25)
            Q3 = df[coluna].quantile(0.75)
            IQR = Q3 - Q1
            limite_inferior = Q1 - 1.5 * IQR
            limite_superior = Q3 + 1.5 * IQR
            return df[(df[coluna] < limite_inferior) | (df[coluna] > limite_superior)]
        
        # Detectando outliers no salário
        outliers_salario = detectar_outliers_iqr(df, "Salario")
        print("Outliers identificados no salário:\n", outliers_salario)
        
        # Removendo os outliers
        df_limpo = df[~df.index.isin(outliers_salario.index)]
        print("\nDataframe após a remoção de outliers:")
        print(df_limpo.describe())
        
        # Normalizando os dados de salário e horas de trabalho (Min-Max Scaling)
        df_limpo["Salario_Normalizado"] = (df_limpo["Salario"] - df_limpo["Salario"].min()) / (df_limpo["Salario"].max() - df_limpo["Salario"].min())
        df_limpo["Horas_Normalizadas"] = (df_limpo["Horas_Trabalho"] - df_limpo["Horas_Trabalho"].min()) / (df_limpo["Horas_Trabalho"].max() - df_limpo["Horas_Trabalho"].min())
        
        # Visualizando os dados antes e depois da remoção de outliers
        plt.figure(figsize=(12, 5))
        plt.subplot(1, 2, 1)
        sns.boxplot(y=df["Salario"])
        plt.title("Distribuição do Salário - Antes da Limpeza")
        
        plt.subplot(1, 2, 2)
        sns.boxplot(y=df_limpo["Salario"])
        plt.title("Distribuição do Salário - Após a Remoção de Outliers")
        
        plt.show()
